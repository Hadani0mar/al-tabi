    BEGIN TRANSACTION -- Start the transaction..

CREATE TABLE #SSS1
(

FREE_L float ,

)


SELECT @QTY_PRICE  = QTY_PRICE ,  @SALE_KEEP_HISTORY =  SALE_KEEP_HISTORY , @UNIT_SOW = UNIT_SOW , @O_OF_STOCK = O_OF_STOCK ,@DISCOUNT_TYPE = DISCOUNT_TYPE ,  @NO_QTY_F7 = NO_QTY_F7 , @COLLECT_TYPE = COLLECT_TYPE , @SERIAL_FLAG = SERIAL_PRINT FROM  SITTEINGS WITH (NOLOCK)

ELSE

SELECT @TOTAL_PRICE_BEFORE = isnull(SUM( (PRICE * QTY) ),0)   FROM SALE_ITEMS_VIEW WITH (NOLOCK)   WHERE  S_ID = @S_ID


select @QTY_FOR_DISCOUNT = QTY ,

@PRICE_QTY_FOR_DISCOUNT = PRICE_QTY ,

@FREE_LEVEL0 = FREE_LEVEL ,
@FREE_QTY0 = FREE_QTY ,

@FREE_LEVEL1 = FREE_LEVEL1 ,
@FREE_QTY1 = FREE_QTY1 ,

@FREE_LEVEL2 = FREE_LEVEL2 ,
@FREE_QTY2 = FREE_QTY2 ,

@FREE_LEVEL3 = FREE_LEVEL3 ,

from BARCODE  WITH (NOLOCK) where item_id = @ITEM_ID and UNIT_ID =@UNIT_ID and @UNIT_QTY = @UNIT_QTY


SET @FREE_LEVEL0 = @FREE_LEVEL0 * @UNIT_QTY

SET @FREE_QTY1 = @FREE_QTY1 * @UNIT_QTY 

SET @FREE_LEVEL2 = @FREE_LEVEL2 * @UNIT_QTY 
SET @FREE_QTY2 = @FREE_QTY2 * @UNIT_QTY 

SET @FREE_LEVEL3 = @FREE_LEVEL3 * @UNIT_QTY 
SET @FREE_QTY3 = @FREE_QTY3 * @UNIT_QTY 

SET @FREE_LEVEL4 = @FREE_LEVEL4 * @UNIT_QTY 
SET @FREE_QTY4 = @FREE_QTY4  * @UNIT_QTY

SET @FREE_QTY5 = @FREE_QTY5 

SET @FREE_LEVEL6 = @FREE_LEVEL6 * @UNIT_QTY 
SET @FREE_QTY6 = @FREE_QTY6  * @UNIT_QTY


SET @QTY = isnull(@QTY  * @UNIT_QTY,0) 
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL0 ,@FREE_QTY0 ,@QTY )

insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL1 ,@FREE_QTY1 ,@QTY)
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL2 ,@FREE_QTY2 ,@QTY)
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL3 ,@FREE_QTY3,@QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL4 ,@FREE_QTY4,@QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL5 ,@FREE_QTY5,@QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL6 ,@FREE_QTY6 ,@QTY)

SELECT @TYPE = TYPE FROM ITEMS WITH (NOLOCK)  WHERE ITEM_ID =@ITEM_ID

set @TOTAL_NOT_DONE = 0
if @CAT3 = 0
Set @CATEOGRY3 = Null

if @PRICE > 0  and @QTY_PRICE = 1
set @QTY_FOR_DISCOUNT = isnull(@QTY_FOR_DISCOUNT  * @UNIT_QTY,0) 
else
set @QTY_FOR_DISCOUNT = 0 ;

begin


 
 IF(@UNIT_QTY > 1)
 BEGIN
 SELECT @UNIT_PRICE = ISNULL(PRICE1, 0 )  FROM BARCODE  WITH (NOLOCK) WHERE ITEM_ID = @ITEM_ID  AND UNIT_QTY = 1

 IF(@UNIT_PRICE = 0 )
 SET @UNIT_PRICE = @PRICE / @UNIT_QTY


 END
 ELSE
 SET @UNIT_PRICE = @PRICE




set @ITEM_ID_F7 = 0

set @STATE_FLAG = 3
else
begin

set @QTY_LEFT1 = 0


if @QTY_FOR_DISCOUNT > 0  
begin

select @PRICE = PRICE1 From BARCODE WITH (NOLOCK)  Where ITEM_ID = @ITEM_ID And UNIT_ID = @UNIT_ID 

end



if @QTY_FOR_DISCOUNT > 0 and @QTY >=@QTY_FOR_DISCOUNT AND (@PRICE >  (@PRICE_QTY_FOR_DISCOUNT / @QTY_FOR_DISCOUNT))
begin
Set @PRICE_2 = @PRICE 

Set @PRICE = @PRICE_QTY_FOR_DISCOUNT / @QTY_FOR_DISCOUNT 

Set @PRICE_3 = @PRICE



select  @PRICEs = PRICE ,  @PRICE_DISC =  PRICE_DISC from CUSTOMERS WITH (NOLOCK)  where  CUST_ID =  @CUST_ID 

if(@PRICEs  =  10 and @PRICE_DISC > 0 )
set  @PRICE = @PRICE + (@PRICE * @PRICE_DISC / 100)
Set @PRICE_3 = @PRICE
 



Set @REMINDER_QTY =( @QTY /@QTY_FOR_DISCOUNT) - floor( @QTY /@QTY_FOR_DISCOUNT) 



set @REMINDER_QTY =  @QTY - (@REMINDER_QTY * @QTY_FOR_DISCOUNT)

set @QTY_LEFT1 = (@QTY - @REMINDER_QTY) 


set @QTY = @REMINDER_QTY
 
end

else
if(@FREE_LEVEL0 > 0 or @FREE_LEVEL1 > 0 or @FREE_LEVEL2 > 0 or @FREE_LEVEL3 > 0 or @FREE_LEVEL4 > 0 or @FREE_LEVEL5 > 0  or @FREE_LEVEL6 > 0)
begin

select @FREE_QTY = max(FREE_q )  from #SSS1 where FREE_L > 0 and QTY >= FREE_L 

select  @FREE_LEVEL = FREE_L  from #SSS1 where FREE_q = @FREE_QTY order by FREE_L desc



if   (@QTY > (floor( @QTY /@FREE_LEVEL ) * @FREE_LEVEL ))
begin

delete from #SSS1


select @R_QTY = @QTY - (floor( @QTY /@FREE_LEVEL ) * @FREE_LEVEL )

insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL0 ,@FREE_QTY0 ,@R_QTY )

insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL1 ,@FREE_QTY1 ,@R_QTY)
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL2 ,@FREE_QTY2 ,@R_QTY)
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL3 ,@FREE_QTY3,@R_QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL4 ,@FREE_QTY4,@R_QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL5 ,@FREE_QTY5,@R_QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL6 ,@FREE_QTY6 ,@R_QTY)

if   (@R_QTY > (floor( @R_QTY /@R_FREE_LEVEL ) * @R_FREE_LEVEL ))

delete from #SSS1

insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL0 , @FREE_QTY0   ,@R11_QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL1 ,@FREE_QTY1 ,@R11_QTY)
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL2 ,@FREE_QTY2 ,@R11_QTY)
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL3 ,@FREE_QTY3 ,@R11_QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL4 ,@FREE_QTY4 ,@R11_QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL5 ,@FREE_QTY5 ,@R11_QTY )
insert into #SSS1 (FREE_L,FREE_q ,QTY ) values (@FREE_LEVEL6 ,@FREE_QTY6 ,@R11_QTY)



select @R11_FREE_QTY = max(FREE_q )  from #SSS1 where FREE_L > 0 and QTY >= FREE_L 


select  @R11_FREE_LEVEL = FREE_L  from #SSS1 where FREE_q = @R11_FREE_QTY order by FREE_L desc


end
